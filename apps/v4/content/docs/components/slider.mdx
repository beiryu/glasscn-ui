## Usage

```tsx
import { Slider } from "@/components/ui/slider"
```

### Default

```tsx
<Slider defaultValue={[50]} max={100} step={1} />
```

### Range

```tsx
<Slider defaultValue={[40, 60]} max={100} step={1} />
```
